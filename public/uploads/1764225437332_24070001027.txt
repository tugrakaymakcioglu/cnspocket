    printf("Choose One Option: \n (1) Add\n (2) Remove Stock\n (3) Show Stock Status\n (4) Exit\n >>>  ");
    scanf("%d", &input);
    return input;
}
int f_add () {
    int value = 0;
    printf("Eklemek istediğiniz miktarı seçin: \n");
    scanf("%d",&value);
    stockAmount = stockAmount + value;
    return stockAmount;
}
int f_remove () {

    int value = 0;
    while (true) {
        printf("Çıkarmak istediğiniz miktarı seçin: \n");

}

}

}
